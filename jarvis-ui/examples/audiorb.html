const ctx = canvas.getContext('2d'); const callButton = document.getElementById('callButton'); const endButton = document.getElementById('endButton'); // Constants const dpr = window.devicePixelRatio || 1; const pi = Math.PI; const points = 12; const baseOrbRadius = 135; // Reduced by 10% (was 150) const canvasSize = 400; // State let currentOrbRadius = baseOrbRadius * 0.9; // Start 10% smaller in idle let isListening = false; let orbSizeMultiplier = 0.9; // Start at 90% size when idle let audioContext = null; let analyser = null; let microphone = null; let animationId = null; let tick = 0; let voiceIntensity = 0; let targetIntensity = 0; // Canvas dimensions - proper scaling const w = canvasSize; const h = canvasSize; canvas.width = w * dpr; canvas.height = h * dpr; canvas.style.width = w + 'px'; canvas.style.height = h + 'px'; // Scale context for DPR ctx.scale(dpr, dpr); // Center point (in CSS pixels, not physical pixels) const center = { x: w / 2, y: h / 2 }; // Swing circles setup const circles = []; for (let idx = 0; idx < 3; idx++) { const swingpoints = []; for (let i = 0; i < points; i++) { const radian = (pi * 2 / points) * i; swingpoints.push({ x: center.x + baseOrbRadius * 0.9 * Math.cos(radian), // Start smaller y: center.y + baseOrbRadius * 0.9 * Math.sin(radian), radian: radian, range: random(5, 15), phase: random(0, pi * 2) }); analyser = audioContext.createAnalyser(); microphone = audioContext.createMediaStreamSource(stream); analyser.fftSize = 256; analyser.smoothingTimeConstant = 0.7; // Slightly less smoothing for faster response analyser.minDecibels = -70; // More sensitive to quieter sounds analyser.maxDecibels = -10; // Adjusted for typical speaking levels microphone.connect(analyser); isListening = true; updateButtonState(); processAudio(); } catch (error) { console.error('Error accessing microphone:', error); analyser.getByteFrequencyData(dataArray); // Calculate average volume let sum = 0; for (let i = 0; i < bufferLength; i++) { sum += dataArray[i]; } const average = sum / bufferLength; // More sensitive normalization - amplify lower volumes // Map average (0-128) to (0-1) with increased sensitivity const rawIntensity = average / 128; }); endButton.addEventListener('click', () => { stopAudio(); }); // Gradient creation function createGradient(angle, colorStops, x, y, radius) { const angleRad = ((angle - 90) * Math.PI) / 180; const startX = x - Math.cos(angleRad) * radius; const startY = y - Math.sin(angleRad) * radius; const endX = x + Math.cos(angleRad) * radius; const endY = y + Math.sin(angleRad) * radius; const gradient = ctx.createLinearGradient(startX, startY, endX, endY); colorStops.forEach(stop => { gradient.addColorStop(stop.position / 100, stop.color); }); return gradient; } // Animation loop function animate() { ctx.clearRect(0, 0, w, h); // Smooth voice intensity transition voiceIntensity += (targetIntensity - voiceIntensity) * 0.1; // Smooth size multiplier transition (grows when listening) if (isListening) { orbSizeMultiplier += (1.0 - orbSizeMultiplier) * 0.08; // Smooth transition to 100% size } else { orbSizeMultiplier += (0.9 - orbSizeMultiplier) * 0.08; // Smooth transition to 90% size } // Calculate orb radius based on voice and listening state const idlePulse = Math.sin(tick * 0.02) * 0.015; // More subtle idle pulse const voicePulse = voiceIntensity * 0.35; // Increased voice reactive pulse (up to 35% increase) const pulseScale = orbSizeMultiplier * (1 + idlePulse + voicePulse); currentOrbRadius = baseOrbRadius * pulseScale; // Draw background circles with gradients circles.forEach((swingpoints, k) => { // Update swing points with organic movement swingpoints.forEach((point, i) => { // Very subtle random movement in idle, more with voice point.phase += random(1, 10) * -0.005 * (1 + voiceIntensity * 4); // Increased voice multiplier // Create phase variation for organic movement let phase = 2 * Math.sin(tick / 85); // More subtle phase // Base radius with organic displacement const baseRadius = currentOrbRadius + 3; // Smaller base offset const displacement = point.range * phase * Math.sin(point.phase) * (0.5 + voiceIntensity * 2.5); // Increased voice effect const r = Math.max(currentOrbRadius * 0.85, Math.min(baseRadius + displacement * 0.8, currentOrbRadius * 1.4)); // Wider range // Very subtle rotation for idle, more with voice point.radian += (pi / 720) * (0.5 + voiceIntensity * 3); // More rotation with voice point.x = center.x + r * Math.cos(point.radian); point.y = center.y + r * Math.sin(point.radian); }); // Set composite operation for each layer if (k === 0) ctx.globalCompositeOperation = 'multiply'; else if (k === 1) ctx.globalCompositeOperation = 'screen'; else ctx.globalCompositeOperation = 'overlay'; // Apply blur ctx.filter = `blur(${1.5 + voiceIntensity * 1.5}px)`; // More blur variation with voice // Create gradient with dynamic opacity based on voice const opacity = 0.08 + voiceIntensity * 0.35; // Higher opacity range for speaking // Clamp gradient radius to prevent clipping const maxGradientRadius = Math.min(w, h) * 0.38; // Slightly larger max const gradientRadius = Math.min(currentOrbRadius * (1.3 + voiceIntensity * 0.6), maxGradientRadius); // More spread with voice // Dynamic gradient angle for each layer const angle = (tick / 150) + (k * Math.PI / 2); // Slower angle change const gradient = createGradient(98, [ { position: 3.79, color: `rgba(20, 186, 203, ${opacity})` }, { position: 27.52, color: `rgba(81, 136, 241, ${opacity * 1.5})` }, { position: 51.25, color: `rgba(176, 102, 205, ${opacity * 2})` }, { position: 74.98, color: `rgba(241, 126, 153, ${opacity * 1.5})` }, { position: 98.71, color: `rgba(245, 149, 147, ${opacity})` } ], center.x, center.y, gradientRadius); } // Draw functions function drawCurve(pts, fillStyle) { ctx.fillStyle = fillStyle; ctx.beginPath(); ctx.moveTo( (pts[pts.length - 1].x + pts[0].x) / 2, (pts[pts.length - 1].y + pts[0].y) / 2 ); for (let i = 0; // Set opacity based on voice ctx.globalAlpha = 0.9 + voiceIntensity * 0.1; ctx.fillStyle = gradient; ctx.beginPath(); ctx.arc(center.x, center.y, currentOrbRadius, 0, pi * 2); ctx.fill(); // Reset ctx.filter = 'none';
